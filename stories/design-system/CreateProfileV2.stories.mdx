import { Meta, Story, Canvas } from '@storybook/addon-docs/blocks';
import CreateProfileV2 from '../../components/CreateProfileV2';

<Meta title="Design system/CreateProfileV2" component={CreateProfileV2} />

<Canvas>
  <Story name="Default">
    {() => {
      const [action, setAction] = React.useState(null);
      const [email, setEmail] = React.useState('');
      return (
        <div>
          <CreateProfileV2
            onPersonalSubmit={data => setAction(`Personal Create data: ${JSON.stringify(data)}`)}
            onOrgSubmit={data => setAction(`Org Create data: ${JSON.stringify(data)}`)}
            onSecondaryAction={() => setAction('Secondary action button was clicked!')}
            onEmailChange={setEmail}
            email={email}
          />
          {action && <p>{action}</p>}
        </div>
      );
    }}
  </Story>
</Canvas>
