import { Meta, Story, Canvas } from '@storybook/addon-docs/blocks';
import styled from 'styled-components';
import { Box } from '../../components/Grid';
import { P } from '../../components/Text';

<Meta title="Design system/Box" component={Box} />

<Canvas>
  <Story name="Using fractions to set percentage width">
    {() => {
      const Block = styled(Box)`
        border: 1px solid black;
      `;
      return (
        <div>
          <Block width={1 / 4} mb={3} p={2}>
            <P>width: 25%;</P>
          </Block>
          <Block width={1 / 3} mb={3} p={2}>
            <P>width: 33%;</P>
          </Block>
          <Block width={1 / 2} mb={3} p={2}>
            <P>width: 50%;</P>
          </Block>
          <Block width={3 / 4} mb={3} p={2}>
            <P>width: 75%;</P>
          </Block>
        </div>
      );
    }}
  </Story>
</Canvas>

<Canvas>
  <Story name="Using fixed pixel width">
    {() => {
      const Block = styled(Box)`
        border: 1px solid black;
      `;
      return (
        <Block width={200} mx="auto" p={2}>
          <P>width: 200px; margin: 0 auto;</P>
        </Block>
      );
    }}
  </Story>
</Canvas>

<Canvas>
  <Story name="Using relative unit widths">
    {() => {
      const Block = styled(Box)`
        border: 1px solid black;
      `;
      return (
        <Block width="20rem" mx="auto" p={2}>
          <P>width: 20rem; margin: 0 auto;</P>
        </Block>
      );
    }}
  </Story>
</Canvas>

<Canvas>
  <Story name="Set responsive width using an array of values, works for padding and margin props as well">
    {() => {
      const Block = styled(Box)`
        border: 1px solid black;
      `;
      return (
        <Block width={[1, 1 / 2, 1 / 4]} mx="auto" p={2}>
          <P>width: responsive; margin: 0 auto;</P>
        </Block>
      );
    }}
  </Story>
</Canvas>
